001package src.pkg_objets;
002import java.util.HashMap;
003import java.util.Map;
004import java.util.Set;
005import java.io.Serializable;
006
007/** 
008 * Class 
009 * @author Group 1, Paris8 University
010 * @version 1.0
011 */
012public class ItemList implements Serializable{
013        private HashMap<String, Objet> objets;
014        private int nombreObjets;
015        public ItemList(){
016                this.objets = new HashMap<String, Objet>();
017        this.nombreObjets = 0;
018        }
019    
020     /**
021     * Gets an object from the instance.
022     * @param ob
023     * @return Object
024     */
025        public Objet getObjet(String ob) {
026        return objets.get(ob);
027    }
028
029    /**
030     * Creates objects.
031     */
032    public void createObjets()
033    {
034        int min = 2;
035        int max = 5;
036        nombreObjets = min + (int)(Math.random() * ((max - min) + 1));// entre 2 à 5
037        for(int  i = 0;i < nombreObjets ; i++){
038            Objet obj = new Objet();
039            objets.put(""+i, obj);
040        }  
041    }
042
043    /**
044     * display the list of objects in the instance.
045     * @return strObjets
046     */
047    public String lookObjets(){
048        String strObjets = "Nombre d'objets: " + nombreObjets + "\n";
049        for(Map.Entry<String, Objet> entry : objets.entrySet()){ 
050            String key = entry.getKey();  
051            Objet obj = entry.getValue();
052            strObjets += key;
053            strObjets += obj;   
054        }
055        return strObjets; 
056    }
057
058    /**
059     * Deletes an object from the instance.
060     * @param ob
061     */
062    public void deleteObjet(String ob){
063        objets.remove(ob);
064        --nombreObjets;
065    }
066
067
068    /**
069     * Adds an object.  
070     * @param ob
071     */
072    public void addObjet(Objet obj){
073        objets.put(""+nombreObjets,obj);
074        nombreObjets++;
075    }
076
077    /**
078     * Checks if the player has a key to 
079     * enter in the room (golden room).
080     * @return boolean
081     */
082    public boolean hasCles(){
083        int i = 0;
084        for(Map.Entry<String, Objet> entry : objets.entrySet()){  
085            Objet obj = entry.getValue();
086            if(obj.getDescription().equals("Clé de la chambre à trésor"))
087                i++;
088        }
089        if (i == 5)
090                return true;    
091        return false; 
092    }
093
094    /**
095     * check the availability of a magic cokier
096     * @param objet ob
097     * @return boolean
098     */
099    public boolean magic(Objet ob){
100        if(ob.getDescription().equals("Un cookie"))
101                return true;
102        return false;
103    }
104
105     /**
106     * check the availability of a beamer
107     * @return boolean
108     */
109    public boolean hasBeamer(){
110        for(Map.Entry<String, Objet> entry : objets.entrySet()){  
111            Objet obj = entry.getValue();
112            if(obj.getDescription().equals("Un beamer"))
113                return true;
114        }
115        return false;
116    }
117}
