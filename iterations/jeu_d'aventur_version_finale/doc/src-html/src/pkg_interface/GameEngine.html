146    /**
147     * Given a command, process (that is: execute) the command. If this command ends
148     * the game, true is returned, otherwise false is returned.
149     * @param commandLine
150     * @return boolean
151     */
152
153    public boolean interpretCommand(String commandLine) {
154        boolean wantToQuit = false;
155        Command command = parser.getCommand(commandLine);
156        CommandWord commandWord = command.getCommandWord();
157        gui.clear();
158        end = (double)System.currentTimeMillis()/1000.0; 
159        time = end - start;
160        gui.time();
161        if( time > timeLimite){
162            return quit();
163        }
164
165        switch(commandWord){
166            case HELP:
167                goRoom(command);
168                break;
169            case LOOK:
170                look();
171                break;
172            case EAT:
173                eat(command);
174                break;
175            case BACK:
176                back(command);
177                break;
178            case GO:
179                goRoom(command);
180                break;
181            case TEST:
182                test(command);
183                break;
184            case TAKE:
185                take(command);
186                break;
187            case DROP:
188                drop(command);
189                break;
190            case ITEMS:
191                items();
192                break;
193            case CHARGER:
194                charger();
195                break;
196            case DECLANCHER:
197                declancher();
198                break;
199            case SAVE:
200                save(command);
201                break;
202            case LOAD:
203                load(command);
204                break;
205            case QUIT:
206                wantToQuit = quit();
207                break;  
208            default : {
209                gui.println("I don't know what you mean...");
210                return false;
211            }
212        }
213        return wantToQuit; 
214    }

228    /**
229     * Tries to go to one direction. If there is an exit, enter the new room,
230     * otherwise prints an error message.
231     * @param command
232     */
233
234    private void goRoom(Command command) {
235        if (!command.hasSecondWord()) {
236            /** 
237             * If there is no second word, we don't know where to go...
238             */
239
240            gui.println("Go where?");
241            return;
242        }
243
244        String direction = command.getSecondWord();
245        Room nextRoom = null;
246        /**
247         * Try to leave current room.
248         */
249
250        if(currentRoom.isTransporterRoom()){
251            nextRoom = getRandomRoom();
252        }else{
253            nextRoom = currentRoom.getExit(direction);
254        }
255
256        if (nextRoom == null) {
257            gui.println("There is no door!");
258        } else {
259            if (("Vous êtes dans la chambre du trésor...".equals(nextRoom.getDescription()))
260                    && !currentPlayer.hasCles()) {
261                gui.println("Oops! Non il faut avoir 5 clés pour accéder à cette chambre");
262                return;
263            }
264            backRoom.push(currentRoom);
265            currentRoom = nextRoom;
266            printLocalInfo();
267            if (currentRoom.getImageName() != null)
268                gui.showImage(currentRoom.getImageName());
269        }
270    }

472    /**
473     * save the state of the game
474     * @param command
475     */
476    public void save(Command command)
477    {
478        if (!command.hasSecondWord()) {
479            gui.println("Save where?");
480            return;
481        }
482        try{
483            String pathName = command.getSecondWord();
484            System.out.println(pathName);
485            FileOutputStream outputFile = new FileOutputStream(pathName);
486            ObjectOutputStream output = new ObjectOutputStream(outputFile);
487            output.writeObject(currentPlayer);
488            output.writeObject(currentRoom);
489            output.writeObject(beamerRoom);
490            output.writeObject(backRoom);
491            output.writeObject(rooms);
492            output.writeObject(time);
493            output.writeObject(start);
494            output.writeObject(end);
495            
496            output.close();
497        }catch(IOException ex){
498            gui.println("File not fond.");
499            gui.println(ex.getMessage());
500        }
501    }

503    /**
504     * load a game state
505     * @param command
506     */
507
508    public void load(Command command)
509    {
510        if (!command.hasSecondWord()) {
511            gui.println("Load where?");
512            return;
513        }
514        try{
515            String pathName = command.getSecondWord();
516            File file = new File(pathName);
517
518            if (!file.exists())
519                return;
520            
521            FileInputStream inputFile = new FileInputStream(file);
522            
523            ObjectInputStream input = new ObjectInputStream(inputFile);
524            this.currentPlayer = (Player)input.readObject();
525            this.currentRoom = (Room)input.readObject();
526            this.beamerRoom = (Room)input.readObject();
527            this.backRoom = (Stack<Room>)input.readObject();
528            this.rooms = (HashMap<String, Room>)input.readObject();
529            this.time = (double)input.readObject();
530            this.start = (double)input.readObject();
531            this.end = (double)input.readObject();
532            printLocalInfo();
533            if (currentRoom.getImageName() != null)
534                gui.showImage(currentRoom.getImageName());
535
536        }catch(IOException ex){
537            gui.println("File not fond.");
538            gui.println(ex.getMessage());
539        }catch(ClassNotFoundException ex){
540            gui.println(ex.getMessage());
541        }
542    }

545    /**
546     * Gets the time.
547     * @return String
548     */
549    public String getTime(){
550        String s = ":";    
551        int t = (int)time;
552        int min = t/60;
553        int second = t - min * 60;
554        if(second < 10)
555            s = ":0";
556
557        return "0"+min+s+second;
558    }
