001package src.pkg_interface;
002
003import javax.swing.*;
004import java.awt.*;
005import java.awt.event.*;
006import java.net.URL;
007import java.awt.image.*;
008
009/**
010 * This class implements a simple graphical user interface with a text entry
011 * area, a text output area and an optional image.
012 * 
013 * @author Michael Kolling
014 * @version 1.0 (Jan 2003)
015 */
016
017public class UserInterface implements ActionListener
018{
019    private GameEngine engine;
020    private JFrame myFrame;
021    private JTextField entryField;
022    private JTextArea log;
023    private JLabel image;
024    private Button buttonLook, buttonNorth, buttonEast, buttonSouth, buttonWest, buttonUp, buttonDown, buttonBack, buttonItems, buttonChrono;
025
026    /**
027     * Constructs a UserInterface. As a parameter, a Game Engine
028     * (an object processing and executing the game commands) is
029     * needed.
030     * 
031     * @param gameEngine  The GameEngine object implementing the game logic.
032     */
033    public UserInterface(GameEngine gameEngine)
034    {
035        engine = gameEngine;
036        createGUI();
037    }
038
039    /**
040     * Prints out some text into the text area.
041     * @param text
042     */
043
044    public void print(String text)
045    {
046        log.append(text);
047        log.setCaretPosition(log.getDocument().getLength());
048    }
049
050    /**
051     * Prints out some text into the text area, followed by a line break.
052     * @param text
053     */
054    public void println(String text)
055    {
056        log.append(text + "\n");
057        log.setCaretPosition(log.getDocument().getLength());
058    }
059    
060    /**
061     * Clears the text printed into the text area.
062     */ 
063
064    public void clear()
065    {
066        log.setText("");
067    }
068    
069
070
071    /**
072     * Shows an image file in the interface.
073     * @param imageName
074     */
075
076    public void showImage(String imageName)
077    {
078        URL imageURL = this.getClass().getClassLoader().getResource(imageName);
079        if(imageURL == null)
080            System.out.println("image not found");
081        else {
082            ImageIcon icon = new ImageIcon(imageURL);
083            image.setIcon(icon);
084            myFrame.pack();
085        }
086    }
087
088    /**
089     * Enable or disable input in the input field.
090     * @param on It's a boolean
091     */
092    public void enable(boolean on)
093    {
094        entryField.setEditable(on);
095        if(!on)
096            entryField.getCaret().setBlinkRate(0);
097    }
098
099    /**
100     * Sets up graphical user interface.
101     */
102
103    private void createGUI()
104    {
105        myFrame = new JFrame(" la cit√© des pharaons.");
106        entryField = new JTextField(34);
107        buttonLook = new Button("look");
108        buttonNorth = new Button("go north");
109        buttonEast = new Button("go east");
110        buttonSouth = new Button("go south");
111        buttonWest = new Button("go west");
112        buttonUp = new Button("go up");
113        buttonDown = new Button("go down");
114        buttonBack = new Button("back");
115        buttonItems = new Button("items");
116            buttonChrono = new Button("00:00");
117        log = new JTextArea();
118        log.setEditable(false);
119        JScrollPane listScroller = new JScrollPane(log);
120        listScroller.setPreferredSize(new Dimension(200, 200));
121        listScroller.setMinimumSize(new Dimension(100,100));
122
123        JPanel panel = new JPanel();
124        JPanel panelBis = new JPanel();
125        panelBis.setLayout(new GridLayout(3, 3));
126        panelBis.add(buttonUp);
127        panelBis.add(buttonNorth);
128        panelBis.add(buttonItems);
129        panelBis.add(buttonWest);
130        panelBis.add(buttonLook);
131        panelBis.add(buttonEast);
132        panelBis.add(buttonBack);
133        panelBis.add(buttonSouth);
134        panelBis.add(buttonDown);
135        
136        image = new JLabel();
137
138        panel.setLayout(new BorderLayout());
139        panel.add(image, BorderLayout.NORTH);
140        panel.add(listScroller, BorderLayout.CENTER);
141        panel.add(entryField, BorderLayout.SOUTH);
142        panel.add(panelBis,BorderLayout.EAST);
143        panel.add(buttonChrono,BorderLayout.WEST);
144        myFrame.getContentPane().add(panel, BorderLayout.CENTER);
145
146        /**
147         * Adds some event listeners to some components.
148         */
149
150        myFrame.addWindowListener(new WindowAdapter() {
151            public void windowClosing(WindowEvent e) {System.exit(0);}
152        });
153
154        entryField.addActionListener(this);
155        buttonUp.addActionListener(this);
156        buttonNorth.addActionListener(this);
157        buttonWest.addActionListener(this);
158        buttonLook.addActionListener(this);
159        buttonEast.addActionListener(this);
160        buttonDown.addActionListener(this);
161        buttonSouth.addActionListener(this);
162        buttonBack.addActionListener(this);
163        buttonItems.addActionListener(this);
164        myFrame.pack();
165        myFrame.setVisible(true);
166        entryField.requestFocus();
167    }
168
169    /**
170     * Actionlistener interface for entry textfield.
171     * @param e ActionEvent
172     */
173    public void actionPerformed(ActionEvent e) 
174    {
175        /**
176         * No need to check the type of action at the moment.
177         * There is only one possible action: text entry.
178         */
179        String composant = e.getActionCommand();
180
181        processCommand(composant);
182    }
183
184    /**
185     * A command has been entered. Reads the command and do whatever is 
186     * necessary to process it.
187     * @param composant
188     */
189
190    private void processCommand(String composant)
191    {
192        String input = composant;
193        
194        boolean finished = false;
195
196        if(finished == true){
197            input = entryField.getText();
198            entryField.setText("");
199        }
200        engine.interpretCommand(input);
201    }
202    
203    /**
204     * Gets the time.
205     */
206
207    public void time(){
208        buttonChrono.setLabel(engine.getTime());
209    }
210
211}
