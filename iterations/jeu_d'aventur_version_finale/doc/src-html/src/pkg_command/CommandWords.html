001package src.pkg_command;
002import java.util.HashMap;
003
004
005/**
006 * This class is part of the "World of Zuul" application. 
007 * "World of Zuul" is a very simple, text based adventure game.
008 * 
009 * This class holds an enumeration of all command words known to the game.
010 * It is used to recognise commands as they are typed in.
011 *
012 * @author  Michael Kolling and David J. Barnes
013 * @version 2006.03.30
014 */
015
016public class CommandWords
017{
018    private HashMap<String, CommandWord> validCommands;
019
020    /**
021     * Constructor - initialise the command words.
022     */
023    public CommandWords()
024    {
025        validCommands = new HashMap<String, CommandWord>();
026        for(CommandWord command : CommandWord.values()) {
027            if(command != CommandWord.UNKNOWN) {
028                validCommands.put(command.toString(), command);
029            }
030        }
031    }
032
033    /**
034     * Find the CommandWord associated with a command word.
035     * @param commandWord The word to look up.
036     * @return CommandWord The CommandWord correspondng to commandWord, or UNKNOWN
037     *         if it is not a valid command word.
038     */
039    public CommandWord getCommandWord(String commandWord)
040    {
041        CommandWord command = validCommands.get(commandWord);
042        if(command != null) {
043            return command;
044        }
045        else {
046            return CommandWord.UNKNOWN;
047        }
048    }
049    
050    /**
051     * Check whether a given String is a valid command word. 
052     * @param String
053     * @return boolean true if it is, false if it isn't.
054     */
055    public boolean isCommand(String aString)
056    {
057        return validCommands.containsKey(aString);
058    }
059
060    /**
061     * Prints all valid commands to System.out.
062     * @return String
063     */
064    public String showAll() 
065    {
066        String str = "";
067        for(String command : validCommands.keySet()) {
068            str += command + "  ";
069        }
070        str+="\n";
071        return str;
072    }
073}
